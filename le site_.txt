  index = (index + 1) % lettres.length;

  document.getElementById('lettre').innerText = lettres[index][0];

// --- Score avec animation et son ---
function updateScore(points){
  score += points;
  const scoreBox = document.getElementById('scoreBox');
  document.getElementById('score').innerText = score;
  scoreBox.style.transform = 'scale(1.5)';
  document.getElementById('winSound').play();
  setTimeout(()=>{ scoreBox.style.transform = 'scale(1)'; },300);
}

// --- Afficher mots ---
function afficherMots(){
  const motCard = document.getElementById('motCard');
  motCard.innerHTML = '';
  mots.forEach(m=>{
    const p = document.createElement('p');
    p.innerHTML = `<span class="lettre"> ${m[0]} </span> ‚Äì <b>${m[1]}</b>`;
    motCard.appendChild(p);
  });
}

// --- Quiz ---
function initQuiz(){
  const quizIndex = Math.floor(Math.random() * lettres.length);
  const quizLettre = lettres[quizIndex];
  document.getElementById('quizLettre').innerText = quizLettre[0];
  document.getElementById('quizImg').src = quizLettre[2];

  const optionsDiv = document.getElementById('options');
  optionsDiv.innerHTML = '';

  let opts = [quizLettre[1]];
  while(opts.length<3){
    const randOpt = lettres[Math.floor(Math.random()*lettres.length)][1];
    if(!opts.includes(randOpt)) opts.push(randOpt);
  }
  opts = opts.sort(()=>Math.random()-0.5);

  opts.forEach(o=>{
    const btn = document.createElement('button');
    btn.className='action';
    btn.innerText=o;
    btn.onclick = ()=>{
      if(o===quizLettre[1]){
        updateScore(10);
        alert('Correct ! ‚≠ê +10 points');
      } else alert('Essaie encore üòä');
      initQuiz();
    };
    optionsDiv.appendChild(btn);
  });
}
